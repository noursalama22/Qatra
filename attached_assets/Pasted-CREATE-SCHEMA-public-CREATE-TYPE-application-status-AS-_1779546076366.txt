CREATE SCHEMA "public";
CREATE TYPE "application_status" AS ENUM('pending', 'approved', 'rejected');
CREATE TYPE "delivery_chain" AS ENUM('humanitarian', 'commercial');
CREATE TYPE "driver_status" AS ENUM('pending', 'active', 'inactive', 'rejected');
CREATE TYPE "driver_task_status" AS ENUM('pending', 'invitation_pending', 'in_progress', 'delivered', 'rejected');
CREATE TYPE "driver_type" AS ENUM('owned', 'independent');
CREATE TYPE "invitation_status" AS ENUM('pending', 'accepted', 'rejected', 'expired');
CREATE TYPE "notification_type" AS ENUM('driver_assigned', 'driver_en_route', 'delivery_complete', 'task_assigned', 'order_placed', 'proof_submitted', 'application_approved', 'application_rejected', 'performance_alert', 'task_invitation', 'route_updated');
CREATE TYPE "order_status" AS ENUM('pending', 'dispatched', 'delivered', 'cancelled');
CREATE TYPE "provider_task_status" AS ENUM('pending', 'acknowledged', 'in_progress', 'delivered');
CREATE TYPE "subscription_frequency" AS ENUM('weekly', 'biweekly', 'monthly');
CREATE TYPE "subscription_status" AS ENUM('active', 'cancelled', 'paused');
CREATE TYPE "task_status" AS ENUM('pending', 'in_progress', 'delivered', 'cancelled');
CREATE TYPE "user_role" AS ENUM('admin', 'ngo', 'provider', 'driver', 'citizen');
CREATE TYPE "zone_status" AS ENUM('active', 'inactive');
CREATE TYPE "payment_related_type" AS ENUM('delivery_order', 'subscription');
CREATE TYPE "payment_status" AS ENUM('pending', 'completed', 'failed', 'refunded');
CREATE TABLE "audit_log" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"actor_id" varchar NOT NULL,
	"actor_role" varchar(50),
	"action" varchar(255) NOT NULL,
	"entity_type" varchar(100) NOT NULL,
	"entity_id" varchar,
	"metadata" jsonb,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "citizens" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"user_id" varchar NOT NULL,
	"zone_id" varchar NOT NULL,
	"lat" numeric(10, 7),
	"lng" numeric(10, 7),
	"created_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "delivery_orders" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"citizen_id" varchar NOT NULL,
	"provider_id" varchar NOT NULL,
	"status" order_status DEFAULT 'pending' NOT NULL,
	"quantity_liters" numeric(10, 2),
	"total_amount" numeric(10, 2),
	"task_id" varchar,
	"mock_payment_token" varchar(255),
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "delivery_proofs" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"task_id" varchar NOT NULL,
	"photo_url" varchar(1024) NOT NULL,
	"signature_url" varchar(1024),
	"notes" text,
	"submitted_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "distribution_tasks" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"ngo_id" varchar NOT NULL,
	"zone_id" varchar NOT NULL,
	"status" task_status DEFAULT 'pending' NOT NULL,
	"quantity_liters" numeric(10, 2) NOT NULL,
	"scheduled_at" timestamp with time zone NOT NULL,
	"assigned_provider_ids" jsonb DEFAULT '[]' NOT NULL,
	"notes" text,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "driver_invitations" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"task_id" varchar NOT NULL,
	"driver_identifier" varchar(255) NOT NULL,
	"driver_id" varchar,
	"status" invitation_status DEFAULT 'pending' NOT NULL,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "driver_tasks" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"driver_id" varchar NOT NULL,
	"provider_ngo_task_id" varchar,
	"order_id" varchar,
	"task_type" delivery_chain NOT NULL,
	"status" driver_task_status DEFAULT 'pending' NOT NULL,
	"zone_id" varchar,
	"provider_id" varchar,
	"scheduled_at" timestamp with time zone,
	"quantity_liters" numeric(10, 2),
	"route_data" jsonb,
	"started_at" timestamp with time zone,
	"delivered_at" timestamp with time zone,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "drivers" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"user_id" varchar NOT NULL,
	"driver_type" driver_type NOT NULL,
	"provider_id" varchar,
	"status" driver_status DEFAULT 'pending' NOT NULL,
	"phone" varchar(50),
	"vehicle_type" varchar(100),
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "gps_positions" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"driver_id" varchar NOT NULL,
	"task_id" varchar,
	"lat" numeric(10, 7) NOT NULL,
	"lng" numeric(10, 7) NOT NULL,
	"accuracy" numeric(8, 2),
	"recorded_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "ngos" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"user_id" varchar NOT NULL,
	"org_name" varchar(255) NOT NULL,
	"contact_email" varchar(255),
	"country" varchar(100),
	"description" text,
	"status" application_status DEFAULT 'pending' NOT NULL,
	"rejection_reason" text,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "notifications" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"user_id" varchar NOT NULL,
	"type" notification_type NOT NULL,
	"message" text NOT NULL,
	"entity_id" varchar,
	"entity_type" varchar(100),
	"read" boolean DEFAULT false NOT NULL,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "payments" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"user_id" varchar NOT NULL,
	"amount" numeric(10, 2) NOT NULL,
	"currency" varchar(10) DEFAULT 'USD' NOT NULL,
	"status" payment_status DEFAULT 'pending' NOT NULL,
	"related_type" payment_related_type,
	"related_id" varchar,
	"mock_payment_token" varchar,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "provider_ngo_tasks" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"distribution_task_id" varchar NOT NULL,
	"provider_id" varchar NOT NULL,
	"status" provider_task_status DEFAULT 'pending' NOT NULL,
	"assigned_driver_id" varchar,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "providers" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"user_id" varchar NOT NULL,
	"company_name" varchar(255) NOT NULL,
	"contact_email" varchar(255),
	"description" text,
	"status" application_status DEFAULT 'pending' NOT NULL,
	"rejection_reason" text,
	"operating_modes" jsonb DEFAULT '[]' NOT NULL,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "sessions" (
	"sid" varchar PRIMARY KEY,
	"sess" jsonb NOT NULL,
	"expire" timestamp NOT NULL
);
CREATE TABLE "signals" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"citizen_id" varchar NOT NULL,
	"zone_id" varchar NOT NULL,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "subscriptions" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"citizen_id" varchar NOT NULL,
	"provider_id" varchar NOT NULL,
	"plan_name" varchar(100) NOT NULL,
	"price_per_delivery" numeric(10, 2),
	"frequency" subscription_frequency,
	"status" subscription_status DEFAULT 'active' NOT NULL,
	"mock_payment_token" varchar(255),
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "system_config" (
	"id" varchar PRIMARY KEY DEFAULT 'default',
	"signal_density_weight" numeric(5, 2) DEFAULT '0.5' NOT NULL,
	"days_since_delivery_weight" numeric(5, 2) DEFAULT '0.3' NOT NULL,
	"population_weight" numeric(5, 2) DEFAULT '0.2' NOT NULL,
	"signal_threshold" integer DEFAULT 10 NOT NULL,
	"platform_fee_percent" numeric(5, 2) DEFAULT '5.0' NOT NULL,
	"auto_escalation_days" integer DEFAULT 3 NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "user_roles" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"user_id" varchar NOT NULL,
	"role" user_role NOT NULL,
	"status" application_status DEFAULT 'approved' NOT NULL,
	"profile_id" varchar,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "users" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"email" varchar CONSTRAINT "users_email_unique" UNIQUE,
	"first_name" varchar,
	"last_name" varchar,
	"profile_image_url" varchar,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
CREATE TABLE "zones" (
	"id" varchar PRIMARY KEY DEFAULT gen_random_uuid(),
	"ngo_id" varchar NOT NULL,
	"name" varchar(255) NOT NULL,
	"description" text,
	"boundary" jsonb,
	"population_estimate" integer,
	"status" zone_status DEFAULT 'active' NOT NULL,
	"signal_count" integer DEFAULT 0 NOT NULL,
	"last_delivery_at" timestamp with time zone,
	"created_at" timestamp with time zone DEFAULT now() NOT NULL,
	"updated_at" timestamp with time zone DEFAULT now() NOT NULL
);
